CREATE DATABASE IF NOT EXISTS hms_batch4;
USE hms_batch4;

DROP TABLE IF EXISTS infant_security;
DROP TABLE IF EXISTS iot_band_monitor;
DROP TABLE IF EXISTS pharmacy_stock;
DROP TABLE IF EXISTS prescriptions;
DROP TABLE IF EXISTS ward_assignments;
DROP TABLE IF EXISTS patients;
DROP TABLE IF EXISTS doctors;
DROP TABLE IF EXISTS access_layers;
DROP TABLE IF EXISTS users;

CREATE TABLE users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(100) NOT NULL UNIQUE,
    password VARCHAR(100) NOT NULL,
    role VARCHAR(50) NOT NULL
);

CREATE TABLE access_layers (
    id INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(100) NOT NULL,
    biothumb_id VARCHAR(100) NOT NULL,
    department VARCHAR(100) NOT NULL,
    FOREIGN KEY (username) REFERENCES users(username)
);

CREATE TABLE doctors (
    id INT AUTO_INCREMENT PRIMARY KEY,
    full_name VARCHAR(120) NOT NULL,
    department VARCHAR(100) NOT NULL,
    availability VARCHAR(100) NOT NULL,
    assigned_ward VARCHAR(100) NOT NULL
);

CREATE TABLE patients (
    id INT AUTO_INCREMENT PRIMARY KEY,
    full_name VARCHAR(120) NOT NULL,
    age INT NOT NULL,
    gender VARCHAR(20) NOT NULL,
    phone VARCHAR(20) NOT NULL,
    patient_condition VARCHAR(150) NOT NULL,
    ward VARCHAR(100) NOT NULL,
    admission_type VARCHAR(50) NOT NULL,
    registered_by VARCHAR(50) NOT NULL,
    emergency_case BOOLEAN NOT NULL,
    admitted_on DATE NOT NULL
);

CREATE TABLE ward_assignments (
    id INT AUTO_INCREMENT PRIMARY KEY,
    patient_name VARCHAR(120) NOT NULL,
    doctor_name VARCHAR(120) NOT NULL,
    nurse_name VARCHAR(120) NOT NULL,
    ward_name VARCHAR(100) NOT NULL,
    room_number VARCHAR(20) NOT NULL,
    schedule_note VARCHAR(120) NOT NULL
);

CREATE TABLE prescriptions (
    id INT AUTO_INCREMENT PRIMARY KEY,
    patient_name VARCHAR(120) NOT NULL,
    doctor_name VARCHAR(120) NOT NULL,
    nurse_name VARCHAR(120) NOT NULL,
    medicines VARCHAR(255) NOT NULL,
    dosage VARCHAR(255) NOT NULL,
    prescribed_on DATE NOT NULL
);

CREATE TABLE pharmacy_stock (
    id INT AUTO_INCREMENT PRIMARY KEY,
    medicine_name VARCHAR(120) NOT NULL,
    stock_in INT NOT NULL,
    stock_out INT NOT NULL,
    balance_units INT NOT NULL,
    prescribed_to VARCHAR(120) NOT NULL,
    prescribed_by VARCHAR(120) NOT NULL,
    approval_status VARCHAR(80) NOT NULL,
    order_status VARCHAR(80) NOT NULL
);

CREATE TABLE iot_band_monitor (
    id INT AUTO_INCREMENT PRIMARY KEY,
    patient_name VARCHAR(120) NOT NULL,
    band_code VARCHAR(80) NOT NULL,
    anomaly VARCHAR(150) NOT NULL,
    heart_rate INT NOT NULL,
    spo2 INT NOT NULL,
    temperature DECIMAL(4,1) NOT NULL,
    monitored_by VARCHAR(120) NOT NULL,
    last_sync VARCHAR(80) NOT NULL
);

CREATE TABLE infant_security (
    id INT AUTO_INCREMENT PRIMARY KEY,
    infant_name VARCHAR(120) NOT NULL,
    mother_name VARCHAR(120) NOT NULL,
    pair_code VARCHAR(80) NOT NULL,
    authorized_members VARCHAR(255) NOT NULL,
    alarm_status VARCHAR(150) NOT NULL,
    alert_targets VARCHAR(255) NOT NULL,
    technical_team VARCHAR(120) NOT NULL
);
